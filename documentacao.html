    /* ========== FUNÇÃO SEGURA: RENDERIZAR PROJETOS ========== */
    function renderProjetos(res) {
        const lista = document.getElementById("lista");
        const statusTxt = document.getElementById("status");
        
        lista.innerHTML = "";
        
        if (!res || !res.sucesso) {
            statusTxt.textContent = "Erro: " + (res?.erro || "Falha ao carregar projetos");
            return;
        }

        const projetos = res.projetos || {};
        const ids = Object.keys(projetos);
        
        statusTxt.textContent = "Projetos encontrados: " + ids.length;
        
        ids.forEach(pid => {
            const p = projetos[pid];
            const card = document.createElement("div");
            card.className = "projetoCard";
            card.textContent = "📁 " + p.name;
            card.onclick = () => abrirProjeto(pid, p);
            lista.appendChild(card);
        });
    }

    /* ========== FUNÇÃO SEGURA: ABRIR PROJETO ========== */
    function abrirProjeto(pid, p) {
        const lista = document.getElementById("lista");
        const statusTxt = document.getElementById("status");
        
        lista.innerHTML = "";
        statusTxt.textContent = "Arquivos do projeto: " + p.name;
        
        const voltar = document.createElement("button");
        voltar.className = "voltarBtn";
        voltar.innerHTML = "⬅️ Voltar para projetos";
        voltar.onclick = () => document.getElementById("loadBtn").click();
        lista.appendChild(voltar);

        const arquivos = p.files || {};
        const nomes = Object.keys(arquivos);
        
        nomes.forEach(nome => {
            const arq = arquivos[nome] || {};
            const link = arq.link || "";
            const content = arq.content || "";
            
            const item = document.createElement("div");
            item.className = "arquivoItem";
            item.innerHTML = "<h3>📄 " + nome + "</h3>";
            
            if (!link && content) {
                item.innerHTML += "<pre style='background:#f5f5f5; padding:10px;'>" + content + "</pre>";
            } 
            else if (link.includes("youtu")) {
                let videoId = "";
                try {
                    if (link.includes("watch")) {
                        videoId = new URL(link).searchParams.get("v");
                    } else {
                        videoId = link.split("/").pop().split("?")[0];
                    }
                } catch {}
                
                if (videoId) {
                    item.innerHTML += ` 
                        <iframe style="width:100%; height:260px; border-radius:6px;" 
                                src="https://www.youtube.com/embed/${videoId}" 
                                allowfullscreen>
                        </iframe>`;
                }
            } 
            else if (link.match(/\.(jpg|jpeg|png|gif|webp)$/i)) {
                item.innerHTML += '<img src="' + link + '" style="max-width:100%; border-radius:6px;">';
            } 
            else if (link.match(/\.(mp4|webm|mov)$/i)) {
                item.innerHTML += '<video controls src="' + link + '" style="max-width:100%; border-radius:6px;"></video>';
            } 
            else if (link.endsWith(".html") || link.endsWith(".htm")) {
                item.innerHTML += '<iframe src="' + link + '" style="width:100%; height:400px; border-radius:6px;"></iframe>';
            } 
            else if (link) {
                item.innerHTML += '<p><small>🔗 Link: <a href="' + link + '" target="_blank">' + link + '</a></small></p>';
            }
            
            lista.appendChild(item);
        });
    }

    /* ========== NOVA FUNÇÃO SEGURA: ATUALIZAR INTERFACE ========== */
    function atualizarInterface() {

            document.getElementById("loginBtn").style.display = "inline-block";
            document.getElementById("logoutBtn").style.display = "none";
            document.getElementById("loadBtn").style.display = "none";
            document.getElementById("status").innerHTML = '🔐 Faça login para ver seus projetos.';
            document.getElementById("lista").innerHTML = '';
        }
    }

    /* ========== EVENTOS SEGUROS ========== */

🔒 Novas Funções de Segurança

1. window.limparURL.getDados()

Função SEGURA que captura os dados do usuário e automaticamente LIMPA a URL, protegendo contra exposição de dados:

// 🔒 USO SEGURO - NÃO expõe dados na URL!
const dados = window.limparURL.getDados();

if (dados.temDados) {
    console.log("Usuário:", dados.userName);
    console.log("Email:", dados.email);
    // A URL já foi limpa automaticamente!
}

2. Evento 'auth-completed'

Evento disparado automaticamente quando um login é detectado:

// 🔒 Detectar login seguro
window.addEventListener('auth-completed', function(e) {
    console.log('Login realizado:', e.detail);
    // e.detail contém { userName, email, ... }
    atualizarInterface();
});

3. window.limparURL.limparDados()

Função para logout seguro:

// 🔒 Logout seguro
document.getElementById("logoutBtn").onclick = function() {
    window.limparURL.limparDados(); // Limpa tudo
    atualizarInterface();
};

📁 API 7 - Sistema de Arquivos

Sistema completo de armazenamento e gerenciamento de arquivos usando tokens:

<script type="module" src="https://code.codehub.ct.ws/API/api7.js"></script>

<script>
const TOKEN = "SEU_TOKEN_AQUI";

// Upload de arquivo
async function uploadFile(file) {
    const reader = new FileReader();
    reader.onload = async () => {
        await api7.upload(TOKEN, {
            name: file.name,
            mimeType: file.type,
            base64: reader.result.split(",")[1]
        });
    };
    reader.readAsDataURL(file);
}

// Listar arquivos
const arquivos = await api7.getAll(TOKEN);

// Renomear
await api7.rename(TOKEN, "id_do_arquivo", "novo_nome.txt");

// Excluir
await api7.remove(TOKEN, "id_do_arquivo");
</script>
